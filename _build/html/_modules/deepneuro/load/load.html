    """Summary
    
    Parameters
    ----------
    dataset : str
        Key for dataset to be returned. E.g., "mets_enhancing"
    output_path: str
        Folder to output loaded data into. If None, will be placed in deepneuro directory.
    
    Returns
    -------
    TYPE
        Description
    """

    # TODO: Loading progress indicator.

    if output_path is None:
        dataset_path = model_dict[dataset][0]

    if not os.path.exists(os.path.dirname(dataset_path)):
        os.makedirs(os.path.dirname(dataset_path))

    if not os.path.exists(dataset_path):
        try:
            urlretrieve(model_dict[dataset][1], dataset_path)
        except Exception:
            if os.path.exists(dataset_path):
                os.remove(dataset_path)
            raise Exception

    return dataset_path
